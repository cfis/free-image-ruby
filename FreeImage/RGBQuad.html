def self.create(red, green, blue, reserved = 0)
  result = self.new
  result[:red] = red
  result[:green] = green
  result[:blue] = blue
  result[:reserved] = reserved
  result
end

def eql?(other)
  other.instance_of?(RGBQuad) and
  self[:red]   == other[:red] and
  self[:green] == other[:green] and
  self[:blue]  == other[:blue] and
  self[:reserved]  == other[:reserved]
end

def icc_profile=(value)
  result = if value
    FreeImage.FreeImage_CreateICCProfile(self, value[:data], value[:size])
  else
    FreeImage.FreeImage_DestroyICCProfile(self)
  end
  FreeImage.check_last_error
  result
end

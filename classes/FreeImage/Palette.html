def [](index)
  unless (0..self.size).include?(index)
    raise(RangeError, "Value is out of range 0..#{self.size}. Value: #{index}")
  end

  ptr = FreeImage.FreeImage_GetPalette(@bitmap)
  FreeImage.check_last_error
  RGBQuad.new(ptr[index])
end
